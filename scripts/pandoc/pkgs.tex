\usepackage{xurl}
% hyperxmp adds the XMP metadata packet that makes the PDFs richly indexable
% (Google Scholar, reference managers), so we keep it. Newer hyperxmp requires
% hyperref to be loaded first, but pandoc emits these header-includes BEFORE it
% loads hyperref — so we defer hyperxmp to end-of-preamble (after hyperref) via
% etoolbox's \AtEndPreamble. etoolbox is loaded explicitly here because pandoc
% loads it later than this include.
\usepackage{etoolbox}
\AtEndPreamble{\usepackage{hyperxmp}}
